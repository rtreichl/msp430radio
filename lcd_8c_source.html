15  // Activate PCA9534

40 
41  // Go in default data mode of display
42  i2c_write_var(PCA9534_I2C_ADR, STOP, 2, 0x02, 0x12);
43 
44  _delay_ten_us(3);
45 
46  lcd_command(LCD_FUNCTION_SET | LCD_2_LINE_INST_1);
47  lcd_command(LCD_BIAS_SET | LCD_BIAS_1_4 | LCD_3_LINE);
48  lcd_command(LCD_POWER_CONTROL | ((0x30 & contrast) >> 4));
49  lcd_command(LCD_FOLLOWER_CONTROL | LCD_FOLLOWER_CIRCUT_ON | LCD_AMPLIFFED_RATIO);
50  lcd_command(LCD_CONTRAST_SET | (0x0F & contrast));
51  lcd_command(LCD_FUNCTION_SET | LCD_INST_0);
52  lcd_command(LCD_DISPLAY_CONTROL | LCD_DISPLAY_ON);
53  lcd_command(LCD_CLEAR_SCREEN);
54  lcd_command(LCD_ENTRY_MODE_SET | LCD_COURSER_INKREMENT);
55  lcd_command(LCD_RESET_COURSER);
56 
57  //Gernarte GCCR Symbole
58  lcd_generatebargraph();
59 
60  //Clear LCD and Init view ram
61  lcd_create_view(0, 0, 0, 0, 2);
62 
63  return 0;
64 }

66 uint8_t lcd_contrast(uint8_t contrast)
67 {
68  lcd_command(LCD_FUNCTION_SET | LCD_2_LINE_INST_1);
69  lcd_command(LCD_POWER_CONTROL | ((0x30 & contrast) >> 4));

71  lcd_command(LCD_FUNCTION_SET | LCD_INST_0);
72 
73  return 0;
