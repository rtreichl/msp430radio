58 void tpa2016d2_equalizer_mode(uint8_t Mode, int8_t Gain)
59 {

62  Gain = 30;
63  if(Gain < -28)
64  Gain = -28;
65  switch(Mode) {
66  case 0:
67  // POP
68  i2c_write_var(I2C_TPA2016D2_ADR, STOP, 7, 0x02, 0x03, 0x08, 0x0A, (Gain & 0x3F), 0x5C, 0xC0);
69  break;
70  case 1:
71  // CLASSIC

73  break;
74  case 2:
75  // JAZZ

78  case 3:

82  case 4:
