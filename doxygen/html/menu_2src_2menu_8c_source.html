96 uint8_t menu_handler(uint8_t *encoder_left_button, int8_t *encoder_left_count, uint8_t *encoder_right_button, int8_t *encoder_right_count)
97 {
98  if (menu_display_selector == 0) {
99  *encoder_left_count = 0;
100  *encoder_left_button = BUTTON_PRESS_FREE;
101  if (*encoder_right_button == BUTTON_PRESS_LONG) {
102  if (actuall_entry->child_long != 0) {
103  actuall_entry = actuall_entry->child_long;
104  }
105  if (actuall_entry->entry_num != 0) {
106  menu_display_selector = actuall_entry->entry_num;
107  menu_function(0, 0, 0, 0);
108  return 0;
109  }
110  *encoder_right_button = BUTTON_PRESS_FREE;
111  }
112  if (*encoder_right_button == BUTTON_PRESS_SHORT) {
113  if (actuall_entry->child_short != 0) {
114  actuall_entry = actuall_entry->child_short;
115  }
116  if (actuall_entry->entry_num != 0) {
117  menu_display_selector = actuall_entry->entry_num;
118  *encoder_right_button = BUTTON_PRESS_FREE;
119  menu_function(0, 0, 0, 0);
120  return 0;
121  }
122  *encoder_right_button = BUTTON_PRESS_FREE;
123  }
124  if (*encoder_right_count != 0) {
125  if(*encoder_right_count > 0) {
126  if (actuall_entry->next != 0) {
127  actuall_entry = actuall_entry->next;
128  }
129  }
130  if (*encoder_right_count < 0) {
131  if (actuall_entry->previous != 0) {
132  actuall_entry = actuall_entry->previous;
133  }
134  }
135  *encoder_right_count = 0;
136  }
137  menu_display();
138  }else {
139  uint8_t ret = menu_function(encoder_left_button, encoder_left_count, encoder_right_button, encoder_right_count);
140  if (ret == 0xFF) { //Menu down wards
141  actuall_entry = actuall_entry->child_short;
142  menu_display_selector = 0;
143  menu_display();
144  } else if (ret == 0xFE) {
145  actuall_entry = actuall_entry->child_long;
146  menu_display_selector = 0;
147  menu_display();
148  } else if (ret == 0xFD) {
149  menu_display_selector = 0;
150  menu_display();
151  }
152  }
153  return 0;
154 }
