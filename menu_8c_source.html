96 uint8_t menu_scroll(uint8_t value)
97 {
98  uint8_t i = 0;
99  char string[2];
100  value = (value * 60) / 100;
101 
102  lcd_create_view("- ", 0, 2, 0, 0);
103  for(i = 1; i <= value / 5 && i <= 12; i++) {
104  lcd_create_view("\5", 1 + i, 2, 0, 0);
105  }
106  if(i <= 12) {
107  string[0] = value % 5;
108  lcd_create_view(string, 1 + i, 2, 0, 0);
109  }
110  i++;
111  for(i = i; i <= 12; i++) {
112  lcd_create_view("\0", 1 + i, 2, 0, 0);
113  }
114  lcd_create_view(" +", 14, 2, 0, 0);
115 
116  return 0;
117 }
