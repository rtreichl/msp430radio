204 uint8_t radio_main(uint8_t *encoder_left_button, int8_t *encoder_left_count, uint8_t *encoder_right_button, int8_t *encoder_right_count, uint8_t entry_num)
205 {
206  uint8_t tmp_value = 0;
207  uint8_t blend_scroll = 0;
208 
209  if(*encoder_right_count != 0) { //move to radio_volume and
210  if(*encoder_right_count < 0 && radio.settings.volume > RADIO_VOLUME_MIN) {
211  radio.settings.volume -= RADIO_VOLUME_STEP;
212  }
213  else if(*encoder_right_count > 0 && radio.settings.volume < RADIO_VOLUME_MAX) {
214  radio.settings.volume += RADIO_VOLUME_STEP;
215  }
216  radio_volume(radio.settings.volume);
217  if(radio.status.audio_status == AUDIO_MUTE) {
218  tpa2016d2_mute(0);
219  }
220  radio.status.audio_status = AUDIO_VOLUME;
221  tmp_value = radio.settings.volume;
222  blend_scroll = 1;
223  *encoder_right_count = 0;
224  }
225 
226  if(*encoder_left_count != 0) {
227  if(*encoder_left_count < 0) {
228  radio.settings.frequency -= RADIO_FREQENCY_STEP;
229  }
230  else if(*encoder_left_count > 0) {
231  radio.settings.frequency += RADIO_FREQENCY_STEP;
232  }
233  radio.status.freq_valid = 0;
234  if(radio.settings.frequency < RADIO_BOT_FREQ){
235  radio.settings.frequency = RADIO_TOP_FREQ;
236  }
237  if(radio.settings.frequency > RADIO_TOP_FREQ){
238  radio.settings.frequency = RADIO_BOT_FREQ;
239  }
240  radio_tune_freq(radio.settings.frequency);
241  tmp_value = ((radio.settings.frequency - RADIO_BOT_FREQ) * 10) / ((RADIO_TOP_FREQ - RADIO_BOT_FREQ) / 10);
242  blend_scroll = 1;
243  *encoder_left_count = 0;
244  }
245  if(*encoder_right_button == BUTTON_SHORT)
246  {
247  *encoder_right_button = BUTTON_FREE;
248  return SHORT_INTO_MENU;
249  }
250  if(*encoder_right_button == BUTTON_LONG)
251  {
252  *encoder_right_button = BUTTON_FREE;
253  return LONG_INTO_MENU;
254  }
255  if(*encoder_left_button == BUTTON_SHORT)
256  {
257  *encoder_left_button = BUTTON_FREE;
258  if(radio.status.audio_status == AUDIO_MUTE) {
259  tpa2016d2_mute(0);
260  radio.status.audio_status = 0;
261  } else {
262  radio.status.audio_status = AUDIO_MUTE;
263  tpa2016d2_mute(1);
264  }
265  }
266  if(*encoder_left_button == BUTTON_LONG)
267  {
268  *encoder_left_button = BUTTON_FREE;
269  //radio_standby
270  }
271 
272  if(radio.settings.ta_tp == 1 && radio.rds.ta == 1 && radio.rds.tp == 1 && radio.status.volume_ta == 0) {
273  radio_volume(radio.settings.volume_ta);
274  radio.status.volume_ta = 1;
275  } else if(radio.rds.ta == 0 && radio.status.volume_ta == 1) {
276  radio_volume(radio.settings.volume);
277  radio.status.volume_ta = 0;
278  }
279 
280  radio_display_handler(blend_scroll, tmp_value);
281  return 0;
282 }

347 uint16_t radio_seeking(uint8_t up_down)
348 {
349  uint8_t resp[8];

351  si4735_fm_seek_start(up_down);
352  ext_interrupt_disable(SI_INT_INT);
353  si4735_fm_tune_status(1, 1, resp);
354 
355  radio.status.freq_valid = 0;
356 
357  return (resp[2] << 8) + resp[3];
